from scipy.stats import norm

# AR(1) state-space regime model
def posterior_update(prior, obs, sigma):
    precision = 1 / prior.var() + 1 / sigma**2
    mu_post   = (prior.mean() / prior.var()
                + obs / sigma**2) / precision
    return norm(mu_post, precision**-.5)
